<html lang="zh-CN">

    <meta charset="UTF-8">
    <title>三种样式表</title>
    <!--1、外部样式表-->
    <link rel="stylesheet" href="notes-css.css">

    <style>
        div {
            color: pink;
        }
    </style>
</head>
<body>
<!--3、行内样式表-->

</body>
</html>
